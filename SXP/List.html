def ==(other)
  case other
    when List
      self.length == other.length && to_a == other.to_a
    when other.respond_to?(:to_list)
      other.to_list == self
    else
      false
  end
end

def each(&block)
  pair = @pair
  while pair != nil
    block.call(pair.head)
    pair = pair.tail
  end
  self
end

def first(count = nil)
  case
    when count.nil?
      @pair.head unless empty?
    when count == 1
      empty? ? [] : [first]
    when count > 1
      empty? ? [] : to_a.first(count)
  end
end

def replace(other_list)
  case other_list
    when Pair
      @pair = other_list
    when List
      @pair = other_list.to_pair
    when Array
      @pair = nil
      unshift(*other_list)
    else
      # TODO
  end
  self
end
